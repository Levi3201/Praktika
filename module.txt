<?php
//подключение к базе данных
$connect = mysqli_connect('localhost', 'root', '', 'brainboost');
//проверка выполнения подключения
if(!$connect)
    die('error');

//принимание данных через метод POST (логин, пароль и почта)
$login = $_POST['login'];
$password = $_POST['password'];
$email = $_POST['email'];

//проверка на корректный адрес электронной почты
if (strpos($email, "@") !== false) {
    //добавление данных в бд, в случае корректного адреса электронной почты
    $data = $connect->query("INSERT INTO `students` (`id`, `login`, `password`, `mail`) 
VALUES (NULL, '$login', '$password', '$email')");
    $_SESSION['message2'] = 'регистрация успешно выполнена, можете войти';
    $connect->close();
    header('Location: ../login.php');
} else {
    //сообщение об ошибке, в сучае некорректного адреса электронной почты
    $_SESSION['message1'] = 'адрес электронной почты имел неверный формат';
    header('Location: ../regist.php');
}
